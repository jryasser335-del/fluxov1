function createCustomPlayer(container) {

    container.innerHTML = `
        <div class="custom-player-wrapper loading">

                            <div class="cp-volume">
                                <button class="cp-btn cp-mute-btn" title="Mute"><i class="fas fa-volume-up"></i></button>

                                    <input type="range" min="0" max="100" value="100" class="cp-volume-range">
                                </div>
                            </div>
                        </div>

                            <button class="cp-btn cp-pip-btn" title="Picture in Picture"><i class="fas fa-clone"></i></button>

                    </div>

            <div class="custom-player-center">

            </div>
            <div class="custom-player-loading"><div class="cp-spinner"></div></div>
            <div class="custom-player-error"><i class="fas fa-exclamation-triangle"></i><p>Stream unavailable</p></div>
        </div>
    `;
    
    const wrapper = container.querySelector('.custom-player-wrapper');
    const video = container.querySelector('#livePlayer');
    const overlay = container.querySelector('.custom-player-overlay');
    const playBtn = container.querySelector('.cp-play-btn');
    const bigPlayBtn = container.querySelector('.custom-player-play');
    const muteBtn = container.querySelector('.cp-mute-btn');
    const volumeRange = container.querySelector('.cp-volume-range');
    const pipBtn = container.querySelector('.cp-pip-btn');
    const fullscreenBtn = container.querySelector('.cp-fullscreen-btn');
    
    // Reset idle timer
    function resetIdle() {
        wrapper.classList.remove('idle');
        clearTimeout(idleTimer);
        idleTimer = setTimeout(() => {
            if (!video.paused) wrapper.classList.add('idle');
        }, 3000);
    }
    
    // Play/Pause
    function togglePlay() {
        if (video.paused) {
            video.play().catch(() => {});
        } else {
            video.pause();
        }
    }
    
    // Update play button icon
    function updatePlayBtn() {
        const icon = video.paused ? 'fa-play' : 'fa-pause';
        playBtn.innerHTML = `<i class="fas ${icon}"></i>`;
        wrapper.classList.toggle('playing', !video.paused);
    }
    
    // Mute toggle
    function toggleMute() {
        video.muted = !video.muted;
        updateVolumeIcon();
    }
    
    // Update volume icon
    function updateVolumeIcon() {
        let icon = 'fa-volume-up';
        if (video.muted || video.volume === 0) icon = 'fa-volume-mute';
        else if (video.volume < 0.5) icon = 'fa-volume-down';
        muteBtn.innerHTML = `<i class="fas ${icon}"></i>`;
    }
    
    // Volume change
    function setVolume(val) {
        video.volume = val / 100;
        video.muted = val === 0;
        volumeRange.value = val;
        updateVolumeIcon();
    }
    
    // Fullscreen toggle - iPhone/iOS compatible
    function toggleFullscreen() {
        // iPhone/iOS requires webkitEnterFullscreen on video element directly
        const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent) || 
                      (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1);
        
        if (!document.fullscreenElement && !document.webkitFullscreenElement && !document.webkitDisplayingFullscreen) {
            // iOS/iPhone: Use native video fullscreen
            if (isIOS && video.webkitEnterFullscreen) {
                try {
                    video.webkitEnterFullscreen();
                    fullscreenBtn.innerHTML = '<i class="fas fa-compress"></i>';
                } catch (e) {
                    console.log('iOS fullscreen error:', e);
                }
            }
            // Standard fullscreen API
            else if (wrapper.requestFullscreen) {
                wrapper.requestFullscreen().catch(e => console.log('Fullscreen error:', e));
                fullscreenBtn.innerHTML = '<i class="fas fa-compress"></i>';
            }
            // Webkit fallback (Safari)
            else if (wrapper.webkitRequestFullscreen) {
                wrapper.webkitRequestFullscreen();
                fullscreenBtn.innerHTML = '<i class="fas fa-compress"></i>';
            }
            // iOS video element fallback
            else if (video.webkitEnterFullscreen) {
                try {
                    video.webkitEnterFullscreen();
                    fullscreenBtn.innerHTML = '<i class="fas fa-compress"></i>';
                } catch (e) {
                    console.log('Video fullscreen error:', e);
                }
            }
        } else {
            // Exit fullscreen
            if (document.exitFullscreen) {
                document.exitFullscreen().catch(e => console.log('Exit fullscreen error:', e));
            } else if (document.webkitExitFullscreen) {
                document.webkitExitFullscreen();
            } else if (video.webkitExitFullscreen) {
                video.webkitExitFullscreen();
            }
            fullscreenBtn.innerHTML = '<i class="fas fa-expand"></i>';
        }
    }
    
    // PiP toggle
    async function togglePiP() {
        try {
            if (document.pictureInPictureElement) {
                await document.exitPictureInPicture();
                pipBtn.classList.remove('active');
            } else if (video.requestPictureInPicture) {
                await video.requestPictureInPicture();
                pipBtn.classList.add('active');
            }
        } catch (e) { console.log('PiP error:', e); }
    }
    
    // Event listeners
    playBtn.addEventListener('click', togglePlay);
    bigPlayBtn.addEventListener('click', togglePlay);
    muteBtn.addEventListener('click', toggleMute);
    volumeRange.addEventListener('input', (e) => setVolume(e.target.value));
    fullscreenBtn.addEventListener('click', toggleFullscreen);
    pipBtn.addEventListener('click', togglePiP);
    
    video.addEventListener('play', updatePlayBtn);
    video.addEventListener('pause', updatePlayBtn);
    video.addEventListener('volumechange', updateVolumeIcon);
    video.addEventListener('playing', () => {
        wrapper.classList.remove('loading');
        wrapper.classList.add('playing');
        resetIdle();
    });
    video.addEventListener('waiting', () => wrapper.classList.add('loading'));
    video.addEventListener('canplay', () => wrapper.classList.remove('loading'));
    video.addEventListener('error', () => wrapper.classList.add('error'));
    
    // Mouse movement for idle detection
    wrapper.addEventListener('mousemove', resetIdle);
    wrapper.addEventListener('touchstart', resetIdle);
    overlay.addEventListener('click', (e) => {
        if (e.target === overlay || e.target.closest('.custom-player-top')) togglePlay();
    });
    
    // Double click for fullscreen
    wrapper.addEventListener('dblclick', (e) => {
        if (!e.target.closest('.cp-btn')) toggleFullscreen();
    });
    
    // Keyboard controls
    wrapper.addEventListener('keydown', (e) => {
        if (e.key === ' ' || e.key === 'k') { e.preventDefault(); togglePlay(); }
        else if (e.key === 'm') toggleMute();
        else if (e.key === 'f') toggleFullscreen();
        else if (e.key === 'ArrowUp') { e.preventDefault(); setVolume(Math.min(100, video.volume * 100 + 10)); }
        else if (e.key === 'ArrowDown') { e.preventDefault(); setVolume(Math.max(0, video.volume * 100 - 10)); }
    });
    
    // Fullscreen change listeners - support all browsers and iOS
    const updateFullscreenIcon = () => {
        const isFs = !!(document.fullscreenElement || document.webkitFullscreenElement || document.webkitDisplayingFullscreen);
        fullscreenBtn.innerHTML = `<i class="fas fa-${isFs ? 'compress' : 'expand'}"></i>`;
    };
    
    document.addEventListener('fullscreenchange', updateFullscreenIcon);
    document.addEventListener('webkitfullscreenchange', updateFullscreenIcon);
    video.addEventListener('webkitbeginfullscreen', updateFullscreenIcon);
    video.addEventListener('webkitendfullscreen', updateFullscreenIcon);
    
    // Make wrapper focusable for keyboard
    wrapper.setAttribute('tabindex', '0');
    
    return video;
}

// Helper function to fetch with retry on rate limit (429)
async function fetchWithRateRetry(url, container, retries = 3) {
    for (let i = 0; i < retries; i++) {
        const res = await fetch(url);
        if (res.status === 429) {
            // Rate limited - wait 3 seconds and retry
            if (container) {
                container.innerHTML = `<div style="display:flex;flex-direction:column;align-items:center;justify-content:center;height:100%;background:#000;color:var(--text-secondary)">
                    <i class="fas fa-clock" style="font-size:3rem;color:var(--accent-5);margin-bottom:16px"></i>
                    <p>Rate limited. Retrying in 3 seconds... (${i + 1}/${retries})</p>
                </div>`;
            }
            await new Promise(r => setTimeout(r, 3000));
            continue;
        }

// Lazy loading for images
let lazyObserver = null;
function initLazyLoading() {
    if (!lazyObserver) {
        lazyObserver = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    const img = entry.target;
                    if (img.dataset.src) {
                        img.src = img.dataset.src;
                        img.removeAttribute('data-src');
                        img.classList.remove('lazy-img');
                        lazyObserver.unobserve(img);
                    }
                }
            });
        }, { rootMargin: '100px' });
    }
    
    document.querySelectorAll('.lazy-img').forEach(img => {
        lazyObserver.observe(img);
    });
}

async function initDB() {
    return new Promise((resolve, reject) => {
        const request = indexedDB.open('TVRecordingsDB', 1);
        request.onupgradeneeded = e => {
            const db = e.target.result;
            if (!db.objectStoreNames.contains('recordings')) {
                db.createObjectStore('recordings', { keyPath: 'id', autoIncrement: true });
            }
        };
        request.onsuccess = e => { db = e.target.result; resolve(); };
        request.onerror = e => reject(e.target.error);
    });
}

    });
}

// IPStreet M3U Integration
async function loadIpstreetData() {
    try {
        const response = await fetch(IPSTREET_M3U_URL);
        if (!response.ok) return;
        const text = await response.text();
        
        const channels = [];
        const lines = text.split('\n');
        let currentChannel = null;
        
        for (let i = 0; i < lines.length; i++) {
            const line = lines[i].trim();
            if (line.startsWith('#EXTINF:')) {
                // Parse channel info
                const nameMatch = line.match(/,([^,]+)$/);
                const name = nameMatch ? nameMatch[1].trim() : 'Unknown';
                const logoMatch = line.match(/tvg-logo="([^"]+)"/);
                const logo = logoMatch ? logoMatch[1] : null;
                const groupMatch = line.match(/group-title="([^"]+)"/);
                const groupTitle = groupMatch ? groupMatch[1] : '';
                
                // Extract country from flag emoji in group title
                let country = null;
                for (const [emoji, code] of Object.entries(FLAG_EMOJI_TO_CODE)) {
                    if (groupTitle.includes(emoji)) {
                        country = code;
                        break;
                    }
                }
                
                // Extract category from group title keywords
                let category = null;
                const groupLower = groupTitle.toLowerCase();
                for (const [keyword, cat] of Object.entries(GROUP_TO_CATEGORY)) {
                    if (groupLower.includes(keyword)) {
                        category = cat;
                        break;
                    }
                }
                
                currentChannel = { 
                    name, 
                    logo,
                    country, 
                    category,
                    groupTitle,
                    iptv_urls: [], 
                    youtube_urls: [], 
                    _source: 'ipstreet' 
                };
            } else if (line && !line.startsWith('#') && currentChannel) {
                currentChannel.iptv_urls.push(line);
                channels.push(currentChannel);
                currentChannel = null;
            }
        }
        
        ipstreetChannels = channels;
        ipstreetDataLoaded = true;
        console.log('IPStreet loaded:', ipstreetChannels.length, 'channels');
    } catch (e) { 
        console.error('IPStreet load failed:', e); 
    }
}

function findIptvCatCountrySlug(code) {
    if (!iptvCatDataLoaded || !iptvCatCountries.length) return null;
    
    const codeLower = code.toLowerCase();
    
    // Special case for "undefined" category
    if (codeLower === 'undefined') {
        const match = iptvCatCountries.find(c => c.name.toLowerCase() === 'undefined');
        return match ? match.url.split('/').pop() : null;
    }
    
    // First try: match by flag URL which contains the country code (most reliable)
    // e.g., flag: "https://iptvcat.com/assets/images/flags/tg.png" for Togo
    let match = iptvCatCountries.find(c => {
        if (!c.flag) return false;
        const flagCode = c.flag.split('/').pop().replace('.png', '').toLowerCase();
        return flagCode === codeLower;
    });
    
    // Second try: special mappings for codes that don't match flag codes
    if (!match) {
        const specialMappings = {
            'gb': 'united_kingdom', 'uk': 'united_kingdom',
            'us': 'united_states_of_america',
            'ru': 'russian_federation',
            'kr': 'south_korea',
            'kp': 'north_korea',
            'cd': 'democratic_republic_of_the_congo',
            'cg': 'republic_of_the_congo'
        };
        const mappedName = specialMappings[codeLower];
        if (mappedName) {
            match = iptvCatCountries.find(c => {
                const slug = c.url.split('/').pop().toLowerCase();
                return slug.startsWith(mappedName);
            });
        }
    }
    
    return match ? match.url.split('/').pop() : null;
}

// Player with Multiple Proxy Fallbacks
async function playWithProxyFallback(url, proxyIndex = 0) {
    if (proxyIndex >= CORS_PROXIES.length) {
        showToast('All proxies failed. Trying direct...', 'error');
        return url; // Return original as last resort
    }
    
    const proxiedUrl = CORS_PROXIES[proxyIndex] + encodeURIComponent(url);
    try {
        const testResponse = await fetch(proxiedUrl, { method: 'HEAD', mode: 'cors' });
        if (testResponse.ok) {
            console.log(`Proxy ${proxyIndex + 1} works:`, CORS_PROXIES[proxyIndex]);
            return proxiedUrl;
        }
    } catch (e) {
        console.log(`Proxy ${proxyIndex + 1} failed, trying next...`);
    }
    return playWithProxyFallback(url, proxyIndex + 1);
}

function getProxiedUrl(url, proxyIdx) {
    if (proxyIdx === 0) return url; // Direct, no proxy
    if (proxyIdx > CORS_PROXIES.length) return url;
    const proxy = CORS_PROXIES[proxyIdx - 1];
    // Custom proxy uses different format
    if (proxy === CUSTOM_PROXY) {
        return proxy + encodeURIComponent(url);
    }
    return proxy + encodeURIComponent(url);
}

    if (Hls.isSupported() && isHLS) {
        // Track errors to prevent infinite retries
        let fragErrorCount = 0;
        let bufferStallCount = 0;
        const MAX_FRAG_ERRORS = 5;
        const MAX_BUFFER_STALLS = 3;
        
        // Optimized HLS config for stable playback
        const hlsConfig = {
            enableWorker: true,
            lowLatencyMode: false,
            
            // Buffer settings - generous buffers to handle stalls
            backBufferLength: 90,
            maxBufferLength: 60,
            maxMaxBufferLength: 120,
            maxBufferSize: 120 * 1000 * 1000,
            maxBufferHole: 1.0, // Allow larger gaps in buffer
            
            // Buffer stall recovery settings
            nudgeOffset: 0.2,
            nudgeMaxRetry: 5,
            highBufferWatchdogPeriod: 3, // Check buffer every 3 seconds
            
            // Live stream settings
            liveSyncDurationCount: 3,
            liveMaxLatencyDurationCount: 10,
            liveDurationInfinity: true,
            liveBackBufferLength: 60,
            
            // Timeouts - more generous for slow connections
            levelLoadingTimeOut: 20000,
            manifestLoadingTimeOut: 20000,
            fragLoadingTimeOut: 30000,
            
            // Retries
            levelLoadingMaxRetry: 3,
            manifestLoadingMaxRetry: 3,
            fragLoadingMaxRetry: 3,
            
            // Performance
            startFragPrefetch: true,
            testBandwidth: true,
            progressive: true,
            
            // ABR settings
            abrEwmaDefaultEstimate: 1000000,
            abrEwmaFastLive: 3,
            abrEwmaSlowLive: 9,
            abrBandWidthFactor: 0.9,
            abrBandWidthUpFactor: 0.6,
            
            // Start settings
            startLevel: -1,
            capLevelToPlayerSize: true,
            
            xhrSetup: function(xhr, xhrUrl) {
                xhr.withCredentials = false;
            }
        };
        
        currentHls = new Hls(hlsConfig);
        
        // Track successful fragment loads to reset error count
        currentHls.on(Hls.Events.FRAG_LOADED, () => {
            fragErrorCount = 0; // Reset on successful load
            bufferStallCount = 0;
        });
        
        currentHls.on(Hls.Events.MANIFEST_PARSED, (event, data) => {
            console.log('Manifest parsed successfully!');
            fragErrorCount = 0;
            bufferStallCount = 0;
            showToast(proxyIdx > 0 ? `Stream loaded via proxy ${proxyIdx}!` : 'Stream loaded!', 'success');
            video.play().catch(e => console.log('Autoplay blocked:', e));
            
            if (data.levels.length > 1) {
                const qualitySelect = document.getElementById('qualitySelect');
                qualitySelect.innerHTML = '<option value="-1">Auto</option>' + 
                    data.levels.map((l, i) => `<option value="${i}">${l.height || Math.round(l.bitrate/1000)}p</option>`).join('');
                document.getElementById('qualityContainer').style.display = '';
            }
        });
        
        currentHls.on(Hls.Events.ERROR, async (event, data) => {
            console.warn('HLS Error:', data.type, data.details, 'fatal:', data.fatal);
            
            // Handle buffer stalls - DON'T restart, just wait
            if (data.details === 'bufferStalledError' || data.details === 'bufferNudgeOnStall') {
                bufferStallCount++;
                console.log(`Buffer stall ${bufferStallCount}/${MAX_BUFFER_STALLS} - waiting for data...`);
                
                // Only take action after multiple stalls
                if (bufferStallCount >= MAX_BUFFER_STALLS) {
                    console.log('Multiple buffer stalls, attempting recovery...');
                    bufferStallCount = 0;
                    
                    // Try to recover by seeking slightly forward
                    if (video.currentTime > 0) {
                        video.currentTime = video.currentTime + 0.1;
                    }
                    currentHls.startLoad();
                }
                return; // Don't treat as fatal
            }
            
            // Handle buffer append errors - try to recover
            if (data.details === 'bufferAppendError' || data.details === 'bufferAppendingError') {
                console.log('Buffer append error, attempting recovery...');
                try {
                    currentHls.recoverMediaError();
                } catch (e) {
                    console.log('Recovery failed:', e);
                }
                return;
            }
            
            // Track fragment load errors
            if (data.details === 'fragLoadError' || data.details === 'fragLoadTimeOut') {
                fragErrorCount++;
                console.log(`Fragment error ${fragErrorCount}/${MAX_FRAG_ERRORS}`);
                
                // Non-fatal fragment errors - HLS.js will retry automatically
                if (!data.fatal) {
                    return;
                }
                
                if (fragErrorCount >= MAX_FRAG_ERRORS) {
                    console.log('Too many fragment errors, trying next source...');
                    currentHls.stopLoad();
                    currentHls.detachMedia();
                    currentHls.destroy();
                    currentHls = null;
                    
                    if (!(await tryNextProxy())) {
                        if (!(await tryNextStream())) {
                            showToast('Stream segments failed to load', 'error');
                            container.querySelector('.custom-player-wrapper')?.classList.add('error');
                        }
                    }
                    return;
                }
            }
            
            // Handle manifest load errors (CORS errors) - try next proxy immediately
            if (data.details === 'manifestLoadError' || data.details === 'manifestParsingError' || 
                data.details === 'levelLoadError' || data.details === 'levelLoadTimeOut') {
                console.log('Manifest/level load error (likely CORS), trying next proxy...');
                currentHls.stopLoad();
                currentHls.detachMedia();
                currentHls.destroy();
                currentHls = null;
                
                if (!(await tryNextProxy())) {
                    if (!(await tryNextStream())) {
                        showToast('Stream failed to load (CORS blocked)', 'error');
                        container.querySelector('.custom-player-wrapper')?.classList.add('error');
                    }
                }
                return;
            }
            
            // Handle fatal errors
            if (data.fatal) {
                switch (data.type) {
                    case Hls.ErrorTypes.NETWORK_ERROR:
                        console.log('Fatal network error, trying next proxy...');
                        currentHls.stopLoad();
                        currentHls.detachMedia();
                        currentHls.destroy();
                        currentHls = null;
                        
                        if (!(await tryNextProxy())) {
                            if (!(await tryNextStream())) {
                                showToast('Network error - stream unavailable', 'error');
                                container.querySelector('.custom-player-wrapper')?.classList.add('error');
                            }
                        }
                        break;
                        
                    case Hls.ErrorTypes.MEDIA_ERROR:
                        console.log('Media error, attempting recovery...');
                        currentHls.recoverMediaError();
                        break;
                        
                    default:
                        // Only give up on truly unrecoverable errors
                        console.log('Unrecoverable error, trying next source...');
                        currentHls.stopLoad();
                        currentHls.detachMedia();
                        currentHls.destroy();
                        currentHls = null;
                        
                        if (!(await tryNextProxy())) {
                            if (!(await tryNextStream())) {
                                showToast('Stream failed', 'error');
                                container.querySelector('.custom-player-wrapper')?.classList.add('error');
                            }
                        }
                        break;
                }
            }
        });
        
        currentHls.loadSource(finalUrl);
        currentHls.attachMedia(video);
        
    } else if (video.canPlayType('application/vnd.apple.mpegurl') && isHLS) {
        // Native HLS (Safari/iOS)
        video.src = finalUrl;
        video.onerror = async () => {
            if (!(await tryNextProxy()) && !(await tryNextStream())) {
                showToast('Stream error', 'error');
                container.querySelector('.custom-player-wrapper')?.classList.add('error');
            }
        };
        video.onloadeddata = () => showToast('Stream loaded!', 'success');
    } else {
        // Direct video file (MP4, WebM, etc.)
        video.src = finalUrl;
        video.onerror = async () => {
            if (!(await tryNextProxy()) && !(await tryNextStream())) {
                showToast('Stream error', 'error');
                container.querySelector('.custom-player-wrapper')?.classList.add('error');
            }
        };
        video.onloadeddata = () => showToast('Stream loaded!', 'success');
    }
}

function toggleFullScreen() {
    const elem = currentPlayer || document.getElementById('playerContainer');
    const video = elem.querySelector('video');
    
    // iPhone/iOS detection
    const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent) || 
                  (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1);
    
    if (!document.fullscreenElement && !document.webkitFullscreenElement && !document.webkitDisplayingFullscreen) {
        // iOS/iPhone: Use native video fullscreen
        if (isIOS && video && video.webkitEnterFullscreen) {
            try {
                video.webkitEnterFullscreen();
            } catch (e) {
                console.log('iOS fullscreen error:', e);
            }
        }
        // Standard fullscreen API
        else if (elem.requestFullscreen) {
            elem.requestFullscreen().catch(e => console.log('Fullscreen error:', e));
        }
        // Webkit fallback (Safari)
        else if (elem.webkitRequestFullscreen) {
            elem.webkitRequestFullscreen();
        }
        // iOS video element fallback
        else if (video && video.webkitEnterFullscreen) {
            try {
                video.webkitEnterFullscreen();
            } catch (e) {
                console.log('Video fullscreen error:', e);
            }
        }
    } else {
        // Exit fullscreen
        if (document.exitFullscreen) {
            document.exitFullscreen().catch(e => console.log('Exit fullscreen error:', e));
        } else if (document.webkitExitFullscreen) {
            document.webkitExitFullscreen();
        } else if (video && video.webkitExitFullscreen) {
            video.webkitExitFullscreen();
        }
    }
}

function startRecording() {
    if (!currentPlayer || !(currentPlayer instanceof HTMLVideoElement)) {
        showToast('Recording only for IPTV', 'error');
        return;
    }
    const stream = currentPlayer.captureStream();
    if (!stream) { showToast('Cannot capture stream', 'error'); return; }
    
    recordedBlobs = [];
    const quality = document.getElementById('recordingQuality').value;
    const opts = { mimeType: 'video/webm; codecs=vp9' };
    if (quality === 'high') opts.videoBitsPerSecond = 2500000;
    else if (quality === 'medium') opts.videoBitsPerSecond = 1000000;
    else if (quality === 'low') opts.videoBitsPerSecond = 500000;
    
    try { mediaRecorder = new MediaRecorder(stream, opts); }
    catch (e) { try { mediaRecorder = new MediaRecorder(stream, { mimeType: 'video/webm' }); } catch (e2) { showToast('Recording not supported', 'error'); return; } }
    
    isRecording = true;
    const btn = document.getElementById('recordBtn');
    btn.classList.add('recording');
    recordingStartTime = Date.now();
    updateRecordingTimer();
    recordingTimerInterval = setInterval(updateRecordingTimer, 1000);
    
    mediaRecorder.ondataavailable = e => { if (e.data?.size > 0) recordedBlobs.push(e.data); };
    mediaRecorder.onstop = () => saveRecording(new Blob(recordedBlobs, { type: 'video/webm' }));
    mediaRecorder.start(10);
    showToast('Recording started', 'success');
}

function stopRecording() {
    if (mediaRecorder?.state !== 'inactive') mediaRecorder.stop();
    isRecording = false;
    clearInterval(recordingTimerInterval);
    const btn = document.getElementById('recordBtn');
    btn.innerHTML = '<i class="fas fa-circle"></i>';
    btn.classList.remove('recording');
}

function updateRecordingTimer() {
    if (!isRecording) return;
    const elapsed = Date.now() - recordingStartTime;
    const s = Math.floor((elapsed / 1000) % 60);
    const m = Math.floor((elapsed / 60000) % 60);
    document.getElementById('recordBtn').textContent = `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
}

function loadMoreChannels() {
    if (isLoadingMore || visibleCount >= displayedChannels.length) return;
    
    isLoadingMore = true;
    const grid = document.getElementById('channelGrid');
    const fragment = document.createDocumentFragment();
    
    const endIndex = Math.min(visibleCount + LOAD_MORE_BATCH, displayedChannels.length);
    
    for (let i = visibleCount; i < endIndex; i++) {
        fragment.appendChild(createChannelCard(displayedChannels[i], i));
    }
    
    grid.appendChild(fragment);
    visibleCount = endIndex;
    
    // Update load more indicator
    updateLoadMoreIndicator();
    
    isLoadingMore = false;
}

function updateLoadMoreIndicator() {
    let indicator = document.getElementById('loadMoreIndicator');
    const remaining = displayedChannels.length - visibleCount;
    
    if (remaining > 0) {
        if (!indicator) {
            indicator = document.createElement('div');
            indicator.id = 'loadMoreIndicator';
            indicator.style.cssText = 'grid-column:1/-1;text-align:center;padding:20px;color:var(--text-secondary);font-size:0.9rem';
            document.getElementById('channelGrid').appendChild(indicator);
        }
        indicator.innerHTML = `<i class="fas fa-spinner fa-spin" style="margin-right:8px"></i>Scroll for ${remaining} more channels...`;
    } else if (indicator) {
        indicator.remove();
    }
}

function setupScrollListener() {
    // Find the scrollable container (main-content or window)
    scrollContainer = document.querySelector('.main-content');
    
    const handleScroll = () => {
        if (!displayedChannels.length || visibleCount >= displayedChannels.length) return;
        
        const scrollElement = scrollContainer || document.documentElement;
        const scrollTop = scrollContainer ? scrollContainer.scrollTop : window.scrollY;
        const scrollHeight = scrollContainer ? scrollContainer.scrollHeight : document.documentElement.scrollHeight;
        const clientHeight = scrollContainer ? scrollContainer.clientHeight : window.innerHeight;
        
        // Load more when near bottom
        if (scrollTop + clientHeight >= scrollHeight - loadMoreThreshold) {
            loadMoreChannels();
        }
    };
    
    // Debounce scroll handler
    let scrollTimeout;
    const debouncedScroll = () => {
        if (scrollTimeout) return;
        scrollTimeout = setTimeout(() => {
            handleScroll();
            scrollTimeout = null;
        }, 100);
    };
    
    if (scrollContainer) {
        scrollContainer.addEventListener('scroll', debouncedScroll, { passive: true });
    }
    window.addEventListener('scroll', debouncedScroll, { passive: true });
}

    if (lastSelected.type === 'livetv') return;

            <button class="filter-pill" data-filter="iptv" onclick="toggleSourceFilter('iptv')"><i class="fas fa-tv"></i> IPTV</button>
            <button class="filter-pill" data-filter="youtube" onclick="toggleSourceFilter('youtube')"><i class="fab fa-youtube"></i> YouTube</button>
            <button class="filter-pill" data-filter="iptvorg" onclick="toggleSourceFilter('iptvorg')"><i class="fas fa-globe"></i> IPTV-org</button>
            <button class="filter-pill" data-filter="iptvcat" onclick="toggleSourceFilter('iptvcat')"><i class="fas fa-cat"></i> IPTV-cat</button>
            <button class="filter-pill" data-filter="ipstreet" onclick="toggleSourceFilter('ipstreet')"><i class="fas fa-street-view"></i> IPStreet</button>
        `;
    }
    
    let source = lastSelected.type === 'favorites' ? favorites : (lastSelected.type === 'all' ? allChannels : currentChannels);
    const search = document.getElementById('searchInput')?.value.toLowerCase().trim() || '';
    const filter = document.querySelector('.filter-pill.active')?.dataset.filter || 'all';
    
    // Use faster filtering with early termination
    const filtered = [];
    for (let i = 0; i < source.length; i++) {
        const ch = source[i];
        
        // Search filter - check name first (most common filter)
        if (search && !ch.name.toLowerCase().includes(search)) continue;
        
        // Source filter
        const isIptvOrg = ch._source === 'iptvorg';
        const isIptvCat = ch._source === 'iptvcat';
        const isIpstreet = ch._source === 'ipstreet';
        
        if (filter !== 'all') {
            const hasIptv = ch.iptv_urls?.length > 0;
            const hasYoutube = ch.youtube_urls?.length > 0;
            
            switch (filter) {
                case 'iptv': if (isIptvOrg || isIptvCat || isIpstreet || !hasIptv) continue; break;
                case 'youtube': if (isIptvOrg || isIptvCat || isIpstreet || !hasYoutube) continue; break;
                case 'iptvorg': if (!isIptvOrg) continue; break;
                case 'iptvcat': if (!isIptvCat) continue; break;
                case 'ipstreet': if (!isIpstreet) continue; break;
            }
        }
        
        filtered.push(ch);
    }
    
    displayChannels(filtered);
}

function toggleSourceFilter(filter) {
    document.querySelectorAll('.filter-pill').forEach(p => p.classList.toggle('active', p.dataset.filter === filter));
    filterAndDisplay();
}

// IPTV-ORG Categories data
let iptvOrgCategories = {};
let iptvOrgCategoriesLoaded = false;

async function loadIptvOrgCategories() {
    if (iptvOrgCategoriesLoaded) return iptvOrgCategories;
    
    try {
        console.log('Loading IPTV-ORG categories...');
        const response = await fetch('https://iptv-org.github.io/iptv/index.category.m3u');
        const m3uText = await response.text();
        
        console.log('M3U loaded, parsing...');
        
        // Parse M3U
        const lines = m3uText.split('\n');
        const categories = {};
        let channelCount = 0;
        
        for (let i = 0; i < lines.length; i++) {
            const line = lines[i].trim();
            
            if (line.startsWith('#EXTINF:')) {
                // Extract metadata
                const groupMatch = line.match(/group-title="([^"]+)"/);
                const nameMatch = line.match(/,(.+)$/);
                const logoMatch = line.match(/tvg-logo="([^"]+)"/);
                const idMatch = line.match(/tvg-id="([^"]+)"/);
                
                // Get the next line which should be the URL
                const nextLine = i + 1 < lines.length ? lines[i + 1].trim() : '';
                
                if (groupMatch && nameMatch && nextLine && (nextLine.startsWith('http://') || nextLine.startsWith('https://'))) {
                    const category = groupMatch[1].toLowerCase(); // Normalize to lowercase
                    const channelName = nameMatch[1];
                    const logo = logoMatch ? logoMatch[1] : '';
                    const id = idMatch ? idMatch[1] : '';
                    const url = nextLine;
                    
                    if (!categories[category]) {
                        categories[category] = [];
                    }
                    
                    categories[category].push({
                        name: channelName,
                        logo: logo,
                        id: id,
                        url: url,
                        _source: 'iptvorg'
                    });
                    
                    channelCount++;
                    i++; // Skip the URL line
                }
            }
        }
        
        console.log(`Parsed ${channelCount} channels into ${Object.keys(categories).length} categories`);
        
        iptvOrgCategories = categories;
        iptvOrgCategoriesLoaded = true;
        return categories;
    } catch (e) {
        console.error('Failed to load IPTV-ORG categories:', e);
        return {};
    }
}

// Fetch with rate limit retry - shows countdown loading bar
async function fetchWithRateLimitRetry(url, maxRetries = 3) {
    for (let attempt = 0; attempt < maxRetries; attempt++) {
        const res = await fetch(url);
        
        if (res.status === 429) {
            // Rate limited - show countdown and retry
            const grid = document.getElementById('channelGrid');
            const waitTime = RATE_LIMIT_WAIT;
            const startTime = Date.now();
            
            // Show countdown loading bar
            grid.innerHTML = `
                <div style="grid-column:1/-1;display:flex;flex-direction:column;align-items:center;justify-content:center;padding:60px 20px;text-align:center">
                    <i class="fas fa-hourglass-half" style="font-size:3rem;color:var(--accent-5);margin-bottom:20px;animation:spin 2s linear infinite"></i>
                    <p style="color:var(--text-primary);font-size:1.1rem;margin-bottom:12px">Rate limited - please wait</p>
                    <p style="color:var(--text-secondary);margin-bottom:20px">Retrying in <span id="retryCountdown">${Math.ceil(waitTime/1000)}</span>s...</p>
                    <div style="width:200px;height:6px;background:rgba(139,92,246,0.2);border-radius:3px;overflow:hidden">
                        <div id="retryProgress" style="width:0%;height:100%;background:var(--gradient-main);border-radius:3px;transition:width 0.1s linear"></div>
                    </div>
                </div>
            `;
            
            // Animate countdown
            await new Promise(resolve => {
                const interval = setInterval(() => {
                    const elapsed = Date.now() - startTime;
                    const remaining = Math.max(0, waitTime - elapsed);
                    const progress = (elapsed / waitTime) * 100;
                    
                    const countdown = document.getElementById('retryCountdown');
                    const progressBar = document.getElementById('retryProgress');
                    
                    if (countdown) countdown.textContent = Math.ceil(remaining / 1000);
                    if (progressBar) progressBar.style.width = `${Math.min(100, progress)}%`;
                    
                    if (elapsed >= waitTime) {
                        clearInterval(interval);
                        resolve();
                    }
                }, 100);
            });
            
            continue; // Retry
        }
        
        return res; // Success or other error
    }
    
    throw new Error('Rate limit exceeded after retries');
}

            const res = await fetchWithRateLimitRetry(DLHD_API);
            const data = await res.json();
            
            if (data.categories) {
                // Filter only sports categories
                const sportsCategories = data.categories.filter(cat => 
                    DLHD_SPORTS_CATEGORIES.includes(cat.name)
                );
                liveChannelsDataDlhd = sportsCategories;
                displayDlhdCategories(sportsCategories);
            } else {
                document.getElementById('channelGrid').innerHTML = '';
                document.getElementById('noResults').style.display = 'block';
                document.getElementById('resultsCount').style.display = 'none';
            }
        } catch (e) {
            console.error('Failed to load DLHD categories:', e);
            showToast('Failed to load DLHD categories', 'error');
            document.getElementById('channelGrid').innerHTML = '';
            document.getElementById('noResults').style.display = 'block';
            document.getElementById('resultsCount').style.display = 'none';
        }
        return;
    }
    
    // Handle CDN Live and All Sources - reset DLHD search state
    currentDlhdData = null;
    currentDlhdType = null;
    
    // Use cached data if available for CDN Live and All Sources
    const cachedData = source === 'cdnlive' ? liveChannelsDataCdn : liveChannelsDataAll;
    if (cachedData.length > 0) {
        liveChannelsData = cachedData;
        displayLiveChannels(cachedData);
        return;
    }
    
    showLoading();
    
    try {
        let data;
        const apiUrl = source === 'cdnlive' ? CDNLIVE_API : LIVE_CHANNELS_API;
        const res = await fetchWithRateLimitRetry(apiUrl);
        data = await res.json();
        
        if (data.success && data.live) {
            liveChannelsData = data.live;
            // Update cache
            if (source === 'cdnlive') {
                liveChannelsDataCdn = data.live;
            } else {
                liveChannelsDataAll = data.live;
            }
            displayLiveChannels(data.live);
        } else {
            liveChannelsData = [];
            document.getElementById('channelGrid').innerHTML = '';
            document.getElementById('noResults').style.display = 'block';
            document.getElementById('resultsCount').style.display = 'none';
        }
    } catch (e) {
        console.error('Failed to load live channels:', e);
        showToast('Failed to load live channels', 'error');
        liveChannelsData = [];
        document.getElementById('channelGrid').innerHTML = '';
        document.getElementById('noResults').style.display = 'block';
        document.getElementById('resultsCount').style.display = 'none';
    }

function openDlhdPlayer(event) {
    if (!event.channels || event.channels.length === 0) {
        showToast('No channels available', 'error');
        return;
    }
    
    // If multiple channels, show selection
    if (event.channels.length > 1) {
        const channelButtons = event.channels.map((ch, idx) => 
            `<button class="control-btn" onclick='playDlhdChannel(${JSON.stringify(ch).replace(/'/g, "\\'")})'>${ch.name}</button>`
        ).join('');
        
        updateMainHeader(`<i class="fas fa-play-circle" style="margin-right:12px;color:var(--accent-3)"></i>${event.title}`, 'Select a channel');
        
        const grid = document.getElementById('channelGrid');
        grid.innerHTML = `
            <div style="grid-column:1/-1;text-align:center;padding:40px">
                <h3 style="margin-bottom:24px;font-size:1.5rem">Select a channel to watch</h3>
                <div style="display:flex;flex-wrap:wrap;gap:12px;justify-content:center">
                    ${channelButtons}
                </div>
            </div>
        `;
    } else {
        playDlhdChannel(event.channels[0]);
    }
}

function closeDlhdPlayer() {
    currentDlhdStreamId = null;
    const dlhdControls = document.getElementById('dlhdPlayerControls');
    if (dlhdControls) {
        dlhdControls.style.display = 'none';
    }
    closePlayer();
}

function switchDlhdPlayer(playerNum) {
    if (!currentDlhdStreamId) return;
    
    const container = document.getElementById('playerContainer');
    const playerUrls = {
        1: `https://dlhd.link/stream/stream-${currentDlhdStreamId}.php`,
        2: `https://dlhd.link/cast/stream-${currentDlhdStreamId}.php`,
        3: `https://dlhd.link/watch/stream-${currentDlhdStreamId}.php`,
        4: `https://dlhd.link/plus/stream-${currentDlhdStreamId}.php`,
        5: `https://dlhd.link/casting/stream-${currentDlhdStreamId}.php`,
        6: `https://dlhd.link/player/stream-${currentDlhdStreamId}.php`
    };
    
    const iframeUrl = playerUrls[playerNum];
    
    container.innerHTML = `
        <iframe 
            src="${iframeUrl}"
            style="width:100%;height:100%;border:none;background:#000"
            allowfullscreen 
            allow="autoplay; encrypted-media; fullscreen; picture-in-picture"
        ></iframe>
    `;
    
    // Update active button in DLHD controls only
    const dlhdControls = document.getElementById('dlhdPlayerControls');
    if (dlhdControls) {
        dlhdControls.querySelectorAll('.control-btn').forEach((btn, idx) => {
            if (idx === playerNum - 1) {
                btn.classList.add('active');
            } else {
                btn.classList.remove('active');
            }
        });
    }
}

function displayLiveTVChannels(reset = false) {
    const grid = document.getElementById('channelGrid');
    const noResults = document.getElementById('noResults');
    const resultsCount = document.getElementById('resultsCount');
    
    if (reset) {
        grid.innerHTML = '';
        liveTVDisplayedCount = 0;
    }
    
    if (!liveTVChannelsData || liveTVChannelsData.length === 0) {
        noResults.style.display = 'block';
        resultsCount.style.display = 'none';
        return;
    }
    
    noResults.style.display = 'none';
    resultsCount.style.display = 'block';
    const channelCountEl = document.getElementById('channelCount');
    if (channelCountEl) {
        channelCountEl.textContent = liveTVChannelsData.length;
    }
    
    // Get next batch
    const startIdx = liveTVDisplayedCount;
    const endIdx = Math.min(startIdx + LIVE_TV_BATCH_SIZE, liveTVChannelsData.length);
    const batch = liveTVChannelsData.slice(startIdx, endIdx);
    
    const html = batch.map((channel, i) => {
        const index = startIdx + i;
        return `
            <div class="channel-card card-hover" style="position:relative" onclick="playLiveTVChannel(${index})">
                ${channel.viewers > 0 ? `<div style="position:absolute;top:16px;right:16px;background:rgba(139,92,246,0.2);color:var(--accent-2);padding:4px 10px;border-radius:20px;font-size:0.7rem;font-weight:700;display:flex;align-items:center;gap:6px"><i class="fas fa-eye"></i>${channel.viewers}</div>` : ''}
                <div class="channel-logo" style="background:rgba(0,0,0,0.3);overflow:hidden">
                    <img src="data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///yH5BAEAAAAALAAAAAABAAEAAAIBRAA7" data-src="${channel.image}" alt="${channel.name}" style="width:100%;height:100%;object-fit:contain" class="lazy-img" onerror="this.parentElement.innerHTML='<i class=\\'fas fa-tv\\' style=\\'font-size:1.8rem;color:var(--accent-1)\\'></i>'">
                </div>
                <div class="channel-name" style="font-size:0.95rem">${channel.name}</div>
                <div class="channel-tags">
                    <span class="channel-tag" style="background:rgba(0,255,204,0.15);color:var(--accent-1);border:1px solid rgba(0,255,204,0.3)">
                        <i class="fas fa-globe"></i> ${channel.code.toUpperCase()}
                    </span>
                </div>
            </div>
        `;
    }).join('');
    
    grid.insertAdjacentHTML('beforeend', html);
    liveTVDisplayedCount = endIdx;
    
    // Initialize lazy loading for new images
    initLazyLoading();
    
    // Add/update load more button
    let loadMoreBtn = document.getElementById('loadMoreLiveTV');
    if (liveTVDisplayedCount < liveTVChannelsData.length) {
        if (!loadMoreBtn) {
            grid.insertAdjacentHTML('afterend', `<div id="loadMoreLiveTV" style="text-align:center;padding:24px"><button class="glow-btn" onclick="displayLiveTVChannels()"><span>Load More (${liveTVChannelsData.length - liveTVDisplayedCount} remaining)</span></button></div>`);
        } else {
            loadMoreBtn.innerHTML = `<button class="glow-btn" onclick="displayLiveTVChannels()"><span>Load More (${liveTVChannelsData.length - liveTVDisplayedCount} remaining)</span></button>`;
        }
    } else if (loadMoreBtn) {
        loadMoreBtn.remove();
    }
}

let lastCdnLiveSourceIndex = null;

async function playLiveSource(sourceIndex) {
    const match = window.currentLiveMatch;
    if (!match) {
        showToast('Match data not found', 'error');
        return;
    }
    
    lastCdnLiveSourceIndex = sourceIndex;
    closeLiveSourcesModal();
    
    const modal = document.getElementById('playerModal');
    modal.classList.add('active');
    
    const source = match.sources[sourceIndex];
    document.getElementById('modalTitle').textContent = `${match.title} - ${source?.name || source?.channelName || source?.source || 'Stream'}`;
    
    // Find Other Source button visibility
    const findBtn = document.getElementById('findOtherSourceBtn');
    if (findBtn) findBtn.style.display = source.type === 'cdnlive' ? 'flex' : 'none';
    
    // Show IPTV controls for HLS sources
    document.getElementById('toggleIPTV').style.display = 'none';
    document.getElementById('toggleYouTube').style.display = 'none';
    document.querySelectorAll('.iptv-control').forEach(el => el.style.display = '');
    
    const container = document.getElementById('playerContainer');
    
    if (source.type === 'kobra') {
        // Kobra sources use embedsports.top - no extraction needed
        document.querySelectorAll('.iptv-control').forEach(el => el.style.display = 'none');
        container.innerHTML = `
            <iframe 
                src="${source.embedUrl}" 
                style="width:100%;height:100%;border:none;background:#000" 
                allowfullscreen 
                allow="autoplay; encrypted-media; fullscreen; picture-in-picture"
            ></iframe>
        `;
    } else if (source.type === 'cdnlive' || source.type === 'titan') {
        // CDN Live and Titan sources - simply load iframe
        document.querySelectorAll('.iptv-control').forEach(el => el.style.display = 'none');
        
        let playerUrl = source.playerUrl || source.url;
        
        // Normalize URL: change user=ntvstream to user=cdnlivetv
        if (playerUrl.includes('cdn-live.tv')) {
            playerUrl = playerUrl.replace(/user=ntvstream/g, 'user=cdnlivetv');
        }
        
        container.innerHTML = `
            <iframe 
                src="${playerUrl}"
                style="width:100%;height:100%;border:none;background:#000"
                allowfullscreen
                allow="autoplay; encrypted-media; fullscreen; picture-in-picture"
            ></iframe>
        `;
    } else {
        container.innerHTML = `<div style="display:flex;flex-direction:column;align-items:center;justify-content:center;height:100%;background:#000;color:var(--text-secondary)">
            <i class="fas fa-exclamation-circle" style="font-size:3rem;color:var(--accent-3);margin-bottom:16px"></i>
            <p>Unknown source type</p>
        </div>`;
    }
}

async function findOtherSource() {
    const match = window.currentLiveMatch;
    if (!match) return;
    
    showToast('Searching DLHD for other sources...', 'success');
    
    const title = match.title;
    const parts = title.split(' vs ');
    const teams = parts.length > 1 ? [title, parts[0].trim(), parts[1].trim()] : [title];
    
    // DLHD Categories to search in order
    const categoriesToSearch = ['Soccer', 'All Soccer Events', 'Sports'];
    
    try {
        // Fetch DLHD all categories first to find encoded names
        const res = await fetch(DLHD_API);
        const data = await res.json();
        if (!data.categories) throw new Error('No DLHD categories');
        
        // Find matching categories
        const matchingCats = [];
        for (const catName of categoriesToSearch) {
            const found = data.categories.find(c => c.name.toLowerCase() === catName.toLowerCase());
            if (found) matchingCats.push(found);
        }
        
        for (const cat of matchingCats) {
            console.log(`Searching in DLHD category: ${cat.name}`);
            const catRes = await fetch(`${DLHD_CATEGORY_API}?name=${cat.encodedName}`);
            const catData = await catRes.json();
            
            if (catData.events) {
                for (const event of catData.events) {
                    const eventTitle = event.title.toLowerCase();
                    // Check if title or any team name matches
                    const isMatch = teams.some(team => {
                        const t = team.toLowerCase();
                        // Precise match for team name (to avoid "Real" matching "Villarreal")
                        // But here we'll use includes for simplicity as requested
                        return eventTitle.includes(t);
                    });
                    
                    if (isMatch) {
                        showToast(`Found alternative on DLHD: ${event.title}`, 'success');
                        
                        if (!event.channels || event.channels.length === 0) continue;
                        
                        if (event.channels.length === 1) {
                            playDlhdChannel(event.channels[0]);
                            return;
                        }
                        
                        // Show selection in player container
                        const container = document.getElementById('playerContainer');
                        container.innerHTML = `
                            <div style="display:flex;flex-direction:column;align-items:center;justify-content:flex-start;height:100%;background:rgba(10,8,25,0.95);padding:20px;text-align:center;overflow-y:auto;-webkit-overflow-scrolling:touch">
                                <h3 style="margin-bottom:10px;font-size:1.2rem;color:#fff">${event.title}</h3>
                                <p style="color:var(--text-secondary);margin-bottom:24px;font-size:0.9rem">Select a language/source:</p>
                                <div style="display:flex;flex-direction:column;gap:10px;width:100%;max-width:500px">
                                    ${event.channels.map(ch => `
                                        <button class="control-btn" style="padding:14px 20px;width:100%;justify-content:center" onclick='playDlhdChannel(${JSON.stringify(ch).replace(/'/g, "\\")})'>
                                            <i class="fas fa-play"></i> ${ch.name}
                                        </button>
                                    `).join('')}
                                </div>
                                <button class="control-btn" style="margin-top:30px;background:rgba(255,255,255,0.05);width:100%;max-width:500px" onclick="playLiveSource(lastCdnLiveSourceIndex)">
                                    <i class="fas fa-undo"></i> Go Back
                                </button>
                            </div>
                        `;
                        return;
                    }
                }
            }
        }
        
        showToast('No alternative sources found on DLHD', 'error');
    } catch (e) {
        console.error('Find other source failed:', e);
        showToast('Search failed', 'error');
    }
}

// Create floating particles
function createParticles() {
    const container = document.getElementById('particles');
    if (!container) return;
    
    const colors = ['#00ffcc', '#8b5cf6', '#ff2d92', '#3b82f6', '#fbbf24'];
    
    for (let i = 0; i < 50; i++) {
        const particle = document.createElement('div');
        const size = Math.random() * 4 + 2;
        const color = colors[Math.floor(Math.random() * colors.length)];
        
        particle.style.cssText = `
            position: absolute;
            width: ${size}px;
            height: ${size}px;
            background: ${color};
            border-radius: 50%;
            left: ${Math.random() * 100}%;
            top: ${Math.random() * 100}%;
            opacity: ${Math.random() * 0.5 + 0.2};
            filter: blur(${Math.random() * 2}px);
            animation: particleFloat ${Math.random() * 20 + 15}s ease-in-out infinite;
            animation-delay: ${Math.random() * -20}s;
        `;
        container.appendChild(particle);
    }
    
    // Add particle animation
    const style = document.createElement('style');
    style.textContent = `
        @keyframes particleFloat {
            0%, 100% { transform: translate(0, 0) rotate(0deg); }
            25% { transform: translate(${Math.random() * 100 - 50}px, ${Math.random() * 100 - 50}px) rotate(90deg); }
            50% { transform: translate(${Math.random() * 100 - 50}px, ${Math.random() * 100 - 50}px) rotate(180deg); }
            75% { transform: translate(${Math.random() * 100 - 50}px, ${Math.random() * 100 - 50}px) rotate(270deg); }
        }
    `;
    document.head.appendChild(style);
}

// TV Remote / Keyboard Navigation
function setupTVNavigation() {
    if (!isSmartTV && !window.matchMedia('(pointer: coarse)').matches) return;
    
    let focusedIndex = 0;
    let focusableElements = [];
    
    function updateFocusables() {
        focusableElements = Array.from(document.querySelectorAll('.channel-card, .sidebar-item, .filter-pill, .control-btn, .modal-close, .promo-close, .glow-btn'));
        focusableElements.forEach((el, i) => {
            el.setAttribute('tabindex', '0');
            el.classList.remove('focused');
        });
        if (focusableElements[focusedIndex]) {
            focusableElements[focusedIndex].classList.add('focused');
            focusableElements[focusedIndex].focus();
        }
    }
    
    function moveFocus(direction) {
        if (focusableElements.length === 0) updateFocusables();
        
        const grid = document.querySelector('.channel-grid');
        const cols = grid ? Math.floor(grid.offsetWidth / (grid.querySelector('.channel-card')?.offsetWidth || 200)) : 4;
        
        let newIndex = focusedIndex;
        switch(direction) {
            case 'up': newIndex = Math.max(0, focusedIndex - cols); break;
            case 'down': newIndex = Math.min(focusableElements.length - 1, focusedIndex + cols); break;
            case 'left': newIndex = Math.max(0, focusedIndex - 1); break;
            case 'right': newIndex = Math.min(focusableElements.length - 1, focusedIndex + 1); break;
        }
        
        if (newIndex !== focusedIndex && focusableElements[newIndex]) {
            focusableElements[focusedIndex]?.classList.remove('focused');
            focusedIndex = newIndex;
            focusableElements[focusedIndex].classList.add('focused');
            focusableElements[focusedIndex].focus();
            focusableElements[focusedIndex].scrollIntoView({ behavior: 'smooth', block: 'center' });
        }
    }
    
    document.addEventListener('keydown', (e) => {
        // Don't intercept keys when typing in an input
        const activeEl = document.activeElement;
        const isTyping = activeEl && (activeEl.tagName === 'INPUT' || activeEl.tagName === 'TEXTAREA' || activeEl.isContentEditable);
        
        if (isTyping && (e.key === 'Backspace' || e.key.length === 1)) {
            return; // Let the input handle it
        }
        
        switch(e.key) {
            case 'ArrowUp': e.preventDefault(); moveFocus('up'); break;
            case 'ArrowDown': e.preventDefault(); moveFocus('down'); break;
            case 'ArrowLeft': e.preventDefault(); moveFocus('left'); break;
            case 'ArrowRight': e.preventDefault(); moveFocus('right'); break;
            case 'Enter': case ' ':
                if (!isTyping) {
                    e.preventDefault();
                    focusableElements[focusedIndex]?.click();
                }
                break;
            case 'Backspace': case 'Escape':
                e.preventDefault();
                const modal = document.querySelector('.modal-overlay.active');
                if (modal) closePlayer();
                else if (document.querySelector('.sidebar.open')) closeMobileSidebar();
                break;
        }
    });
    
    // Update focusables when content changes
    const observer = new MutationObserver(() => setTimeout(updateFocusables, 100));
    observer.observe(document.body, { childList: true, subtree: true });
    
    setTimeout(updateFocusables, 500);
}

document.addEventListener('DOMContentLoaded', () => {
    init();
    // Particles disabled for performance
    setupTVNavigation();
